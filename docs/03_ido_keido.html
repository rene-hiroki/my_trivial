3.4 地図にプロットする

# plot
geo <- df %>% leaflet() %>% addTiles()

for (gru in df$group){
  geo <- geo %>% 
  addMarkers(lng = ~lon, lat = ~lat, popup = ~popup, group = gru, 
             data = dplyr::filter(.data = df, group == gru)) %>% 
  addLayersControl(overlayGroups = df$group,
                   options = layersControlOptions(collapsed = FALSE))  
}
